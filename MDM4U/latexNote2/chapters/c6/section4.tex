\section{Approximate Hypergeometric}
If the population size $N$ is fairly large, and the sample size is relatively small, we can use a \underline{Normal Distribution}
to simulate \underline{Hypergeometric}

\begin{theorem}
    The distribution of a discrete random variable X is a hypergeometric probability distribution if:
    \begin{enumerate}
        \item the probability experiment is repeated a \textbf{fixed} number of times
        \item the outcome of each trial can be categorized into \textbf{success} or failed outcomes
        \item the random varaible $X$ represents the number of \textbf{successes}
        \item each trial of experiment is \textbf{dependant}
    \end{enumerate}
\end{theorem}

\begin{theorem}
    Given that $N$ is the population size, $n$ is the sample size, and $a$ is the number of elements in the population 
    with the particular characteristic, if
    \begin{gather*}
       \frac{n}{N} < \tfrac{1}{10}\\
       n\frac{a}{N} > 5\\
       n(\frac{N - a}{N}) > 5
    \end{gather*}
    then the hypergeometric random variable can be approximated by a normal distribution with 
    \begin{gather*}
        \mu = \frac{an}{N}\\
        \sigma = \sqrt{
            n(\frac{a}{N})(\frac{N - a}{N})(\frac{N - n}{N - 1})
        }
    \end{gather*}
\end{theorem}
