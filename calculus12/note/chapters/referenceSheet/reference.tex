\subsubsection{Log Laws}
\begin{gather*}
    \log_b A + \log_b B = \log_b (A \times B)\\
    \log_b (\tfrac{M}{N}) = \log_b M - \log_b N \\
    log_b M^k = k\log_b M\\
    log_b b^k = k\\
    b^{\log_b k} = k
\end{gather*}
