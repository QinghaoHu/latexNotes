\subsubsection{Properties of Three-Dimensional vector}
Scalar Multiplicatoin:\\
Given $\vec{u} = \left[u_1, u_2, u_3\right]$ and $k \in \R$, then 
$$k\left[u_1, u_2, u_3\right] = \left[ku_1, ku_2, ku_3\right]$$

\noindent Vector Addition:\\
Given $\vec{u} = \left[u_1, u_2, u_3\right]$ and $\vec{v} = \left[v_1, v_2, v_3\right]$, then 
$$\vec{u} + \vec{v} = \left[u_1 + v_1, u_2 + v_2, u_3 + v_3\right]$$

\noindent Vector Subtraction:\\
Given $\vec{u} = \left[u_1, u_2, u_3\right]$ and $\vec{v} = \left[v_1, v_2, v_3\right]$, then 
$$\vec{u} - \vec{v} = \left[u_1 - v_1, u_2 - v_2, u_3 - v_3\right]$$

\noindent Vector's Magnitude:\\
Given $\vec{u} = \left[u_1, u_2, u_3\right]$ , then 
$$\left|\vec{u}\right| = \sqrt{{u_1}^2 + {u_2}^2 + {u_3}^2}$$

\noindent Dot Product:\\
Given $\vec{u} = \left[u_1, u_2, u_3\right]$ and $\vec{v} = \left[v_1, v_2, v_3\right]$, then 
$$\vec{u} \cdot \vec{v} = u_1v_1 + u_2v_2 + u_3v_3$$

\subsubsection{Colinear and Orthogonal Vectors}
Given $\vec{a} \cdot \vec{b} = \left|\vec{a}\right|\left|\vec{b}\right|\cos \theta$ then 
$\cos \theta = \frac{\vec{a} \cdot \vec{b}}{\left|\vec{a}\right|\left|\vec{b}\right|}$

Vectors are considered to be colinear if $\cos \theta = 1$ \\

Vectors are considered to be arthogonal if $\cos \theta = 0$
