\subsection{The Concept of Limit}

\begin{definition}[Limit]
Let $f$ be a function and let $a$ be a real number. If
\[
  \lim_{x\to a}f(x)=L,
\]
then in $\varepsilon$-$\delta$ language:
\[
  \forall\varepsilon>0,\ \exists\delta>0\ \text{such that}\ 0<|x-a|<\delta \implies |f(x)-L|<\varepsilon.
\]
Note that $f(a)$ does not need to be defined.

A two-sided limit exists when:
\begin{itemize}
  \item $\lim_{x\to a^-}f(x)$ exists,
  \item $\lim_{x\to a^+}f(x)$ exists,
  \item $\lim_{x\to a^-}f(x)=\lim_{x\to a^+}f(x)$.
\end{itemize}
\end{definition}

\begin{definition}[Continuity]
A function is continuous at $x=a$ if:
\begin{enumerate}
  \item $\lim_{x\to a}f(x)$ exists,
  \item $f(a)$ is defined,
  \item $\lim_{x\to a}f(x)=f(a)$.
\end{enumerate}
\end{definition}
