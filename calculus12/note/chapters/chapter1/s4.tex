\subsection{Continuity and Limit Laws}

\begin{definition}[Limit Laws]
Assume $\lim_{x\to a}f(x)$ and $\lim_{x\to a}g(x)$ both exist, and let $c$ be a constant. Then:
\begin{enumerate}
  \item $\lim_{x\to a}c = c$
  \item $\lim_{x\to a}x = a$
  \item $\lim_{x\to a}[f(x)+g(x)] = \lim_{x\to a}f(x)+\lim_{x\to a}g(x)$
  \item $\lim_{x\to a}[f(x)-g(x)] = \lim_{x\to a}f(x)-\lim_{x\to a}g(x)$
  \item $\lim_{x\to a}[f(x)\cdot g(x)] = \left(\lim_{x\to a}f(x)\right)\left(\lim_{x\to a}g(x)\right)$
  \item $\lim_{x\to a}\dfrac{f(x)}{g(x)} = \dfrac{\lim_{x\to a}f(x)}{\lim_{x\to a}g(x)}$, provided $\lim_{x\to a}g(x)\ne 0$
  \item $\lim_{x\to a}[f(x)]^n = \left(\lim_{x\to a}f(x)\right)^n$
  \item $\lim_{x\to a}\sqrt[n]{f(x)} = \sqrt[n]{\lim_{x\to a}f(x)}$ (where defined)
\end{enumerate}
\end{definition}

Some cases require extra care.

\begin{example}
Evaluate
\[
  \lim_{x\to 0}\sqrt{x}.
\]
Since $\sqrt{x}$ is only defined for $x\ge 0$, the left-hand limit at $0$ is not defined in $\R$. Therefore the two-sided limit does not exist.
\end{example}

\begin{example}
Evaluate
\[
  \lim_{x\to 1}(x^2-5x).
\]
By substitution,
\[
  \lim_{x\to 1}(x^2-5x)=1^2-5(1)=-4.
\]
\end{example}
