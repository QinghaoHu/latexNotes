\subsection{Discontinuity and Limits}

\subsubsection{Jump Discontinuity}
\begin{example}
Evaluate the limit of
\[
  f(x)=
  \begin{cases}
    \sqrt{x-3}, & x\ge 7, \\
    x-4, & x<7.
  \end{cases}
\]
Then
\[
  \lim_{x\to 7^-}f(x)=3,
  \qquad
  \lim_{x\to 7^+}f(x)=2.
\]
Since the one-sided limits are not equal, $\lim_{x\to 7}f(x)$ does not exist. This is a jump discontinuity.
\end{example}

\subsubsection{Infinite Discontinuity}
\begin{example}
Evaluate the behavior of
\[
  f(x)=\frac{x+3}{x-2}
\]
near $x=2$.
\[
  \lim_{x\to 2^-}f(x)=-\infty,
  \qquad
  \lim_{x\to 2^+}f(x)=+\infty.
\]
The two-sided limit does not exist.
\end{example}

\subsubsection{Removable Discontinuity}
\begin{example}
Evaluate
\[
  \lim_{x\to 5}\frac{2x^2-7x-15}{x-5}.
\]
\begin{align*}
  \lim_{x\to 5}\frac{2x^2-7x-15}{x-5}
  &= \lim_{x\to 5}\frac{(2x+3)(x-5)}{x-5} \\
  &= \lim_{x\to 5}(2x+3) \\
  &= 13.
\end{align*}
The original function is undefined at $x=5$, so this is a removable discontinuity.
\end{example}

\begin{definition}[Indeterminate Form]
An indeterminate form is an expression such as $\frac{0}{0}$ or $\frac{\infty}{\infty}$. Direct substitution in limits can produce these forms, and additional algebraic work is required.
\end{definition}
