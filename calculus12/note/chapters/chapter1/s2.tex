\subsection{Measuring Rates of Change from a Defined Relation}

\subsubsection{Difference Quotient}
\begin{definition}[Difference Quotient]
The difference quotient gives the average rate of change of a function over an interval:
\[
  \frac{f(x+h)-f(x)}{h}, \quad h\ne 0.
\]
It is the slope of the secant line through $(x,f(x))$ and $(x+h,f(x+h))$.
\end{definition}

\begin{proof}
Using the slope formula,
\[
  m=\frac{y_2-y_1}{x_2-x_1},
\]
substitute $(x,f(x))$ and $(x+h,f(x+h))$:
\begin{align*}
  m &= \frac{f(x+h)-f(x)}{(x+h)-x} \\
    &= \frac{f(x+h)-f(x)}{h}.
\end{align*}
\end{proof}

\begin{example}
Find a general expression for the slope when $f(x)=-x^2+4x+5$.
\begin{align*}
  m &= \frac{f(x+h)-f(x)}{h} \\
    &= \frac{-(x+h)^2+4(x+h)+5-(-x^2+4x+5)}{h} \\
    &= \frac{-2xh-h^2+4h}{h} \\
    &= -2x-h+4, \quad h\ne 0.
\end{align*}
\end{example}

\subsubsection{Approximating a Tangent Slope}
If $h$ is very small, the difference quotient
\[
  \frac{f(x+h)-f(x)}{h}
\]
approximates the slope of the tangent line.

Later, limits explain why this approximation works.
