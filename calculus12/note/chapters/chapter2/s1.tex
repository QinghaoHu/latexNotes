\subsection{Derivatives of Polynomial Functions}

\subsubsection{Power Functions}
A power function has the form
\[
  x^a, \quad x\in\R,\ a\in\R,
\]
where $x$ is the base and $a$ is the exponent.

\subsubsection{Power Rule}
If
\[
  f(x)=x^n,
\]
then
\[
  f'(x)=nx^{n-1}.
\]

In words: move the exponent to the front as a coefficient, then subtract 1 from the exponent.

\begin{remark}
Do not apply the power rule to exponential functions such as $a^x$ (for example, $2^x$).
\end{remark}

\subsubsection{Basic Derivative Rules}
\begin{align*}
\text{If } f(x)=c, &\quad f'(x)=0, \\
\text{If } f(x)=x^n, &\quad f'(x)=nx^{n-1}, \\
\text{If } f(x)=c\,g(x), &\quad f'(x)=c\,g'(x), \\
\text{If } h(x)=f(x)+g(x), &\quad h'(x)=f'(x)+g'(x), \\
\text{If } h(x)=f(x)-g(x), &\quad h'(x)=f'(x)-g'(x).
\end{align*}
