\section{The law of the Conservation of Energy}
\subsection{Boring Definitions}
\begin{definition}[Law of the Conservation of Energy]
     Energy will neither be created or destroy, only change from one form to another
\end{definition}

\begin{remark}
    The law of the conservation of energy only works in a \textbf{closed isiolated} system. In reality, the only true \textbf{closed and isolated} system is the Universe
\end{remark}

\subsubsection*{Mechancial Energy}
\begin{definition}[Mechancial Energy]
      The sum of the \textbf{kinetic energy} and \textbf{gravational potential energy}.
\end{definition}

Mathematically:
\begin{equation}
    E_m = E_g + E_k
\end{equation}
