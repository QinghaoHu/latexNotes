\section{Sequence and Series}
\subsection{Arithemtic Sequences and Series}
\begin{theorem}
    [Sum of Arithemtic Sequences and Series]
    Assume we have a sequence with $n$ terms: 
    \[  
        a + (a + d) + (a + 2d) + (a + (n - 1)d)
    \]
    The sum of the sequence is 
    \[
        \frac{2a + (n - 1)d}{2} * n
    \]
\end{theorem}

\subsection{Geometric Sequences and Series}
\begin{theorem}
    [Sum of Geometric Sequences and Series]
    Assume we have a geometric sequence with $n$ terms:
    \[
        a + ar + \cdots + ar^{n - 1} = {{a(r^n - 1)} \over r - 1}
    \]
    The sum of the geometric sequence is
    \[
        \frac{a(r^n - 1)}{r - 1}  \text{ or } \frac{a(r^n - 1)}{1 - r}
    \]

    If a geometric sequence has infinite amount of terms and $\left|r\right| \leq 1$, the sum of this sequence 
    can be expressed as 
    \[
        \frac{a}{1-r}
    \]
\end{theorem}
